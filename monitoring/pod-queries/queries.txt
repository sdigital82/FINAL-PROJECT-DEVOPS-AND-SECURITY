MEMORY PANEL
===============

kube_pod_container_resource_requests{pod="$pod", resource="memory"}

sum(
  container_memory_working_set_bytes{
    namespace="$namespace",
    pod="$pod",
    container!="POD"
  }
)

CPU PANEL
==========
Limits
sum(
  kube_pod_container_resource_limits{
    namespace="$namespace", 
    pod="$pod", 
    resource="cpu",
    container!="POD"
  }
)

Requests:
sum(
  kube_pod_container_resource_requests{
    namespace="$namespace", 
    pod="$pod", 
    resource="cpu",
    container!="POD"
  }
)

Usage:
sum(
  rate(container_cpu_usage_seconds_total{
    namespace="$namespace",
    pod="$pod",
    container!="POD"
  }[2m])
)

Scale: linear
units: SI short
